from pydantic import BaseModel, ValidationErrorimport jsonclass ResponseWithConfidence(BaseModel):    answer: str    confidence: float    sources: list[str]    alternative_interpretations: list[str] = []ERROR_RECOVERY_PROMPT = """Answer the question based on the context provided.Context: {context}Question: {question}Instructions:1. If you can answer confidently (>0.8), provide a direct answer2. If you're somewhat confident (0.5-0.8), provide your best answer with caveats3. If you're uncertain (<0.5), explain what information is missing4. Always provide alternative interpretations if the question is ambiguousRespond in JSON:{{    "answer": "your answer or 'I cannot determine this from the context'",    "confidence": 0.0-1.0,    "sources": ["relevant context excerpts"],    "alternative_interpretations": ["if question is ambiguous"]}}"""async def answer_with_fallback(    context: str,    question: str,    llm) -> ResponseWithConfidence:    """Answer with error recovery and fallback."""    prompt = ERROR_RECOVERY_PROMPT.format(context=context, question=question)    try:        response = await llm.ainvoke(prompt)        return ResponseWithConfidence(**json.loads(response.content))    except (json.JSONDecodeError, ValidationError) as e:        # Fallback: try to extract answer without structure        simple_prompt = f"Based on: {context}\n\nAnswer: {question}"        simple_response = await llm.ainvoke(simple_prompt)        return ResponseWithConfidence(            answer=simple_response.content,            confidence=0.5,            sources=["fallback extraction"],            alternative_interpretations=[]        )

# Pro Tips
- 💡Always start with a clear system prompt to define the AI's persona and task constraints before introducing user instructions.
- 💡For critical applications, combine Chain-of-Thought with self-reflection or validation steps to improve output accuracy and reduce hallucinations.
- 💡Leverage structured outputs (e.g., JSON mode) whenever possible to ensure parseable and reliable data extraction from LLM responses.
